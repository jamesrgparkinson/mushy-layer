87  virtual void computeResidual(Real& a_residual, const Real a_ghost_enthalpy, const Real a_ghost_bulk_concentration, const Real a_interior_temperature)
88  {
89  Real a_ghost_T = m_params.computeTemperature(a_ghost_enthalpy, a_ghost_bulk_concentration);
90 
91  Real Tflux = (a_ghost_T - a_interior_temperature)/m_dx;
92  Real Tboundary = (a_ghost_T + a_interior_temperature)/2;
93 
94  a_residual = (m_F - m_b*(Tboundary - m_T_ref)) - m_a*Tflux;
95 
96  // Some basic error checking
97  // should never have a residual this big?
98  if (abs(a_residual) > 1e3)
99  {
100  MayDay::Warning("NonlinearTemperatureBCRobin::computeResidual - BC residual too large");
101  a_residual = 0.0;
102  }

177  const Real a_ghost_bulk_conc)
178  {
179  CH_TIME("NonlinearBCSolverPicard::solve");
180 
181  Real resid = 0.0;
182  m_bc->computeResidual(resid,
183  ghost_enthalpy, // enthalpy in ghost cell
184  a_ghost_bulk_conc, // bulk conc in ghost cell
185  a_interior_temperature // temperature at interior grid cell
186  );
187 
188  // Compute new estimate for the enthalpy in the ghost cell, by incrementing by the residual
189  ghost_enthalpy = ghost_enthalpy + resid;
190 
191  // 10^-2 is some arbitrary criteria
192  int num_iter = 0;
193  while (abs(resid) > m_max_resid && num_iter < m_max_iter)
194  {
195 
196  // Recompute residual
197  m_bc->computeResidual(resid,
198  ghost_enthalpy, // enthalpy in ghost cell
199  a_ghost_bulk_conc, // bulk conc in ghost cell
200  a_interior_temperature // temperature at interior grid cell
201  );
202 
203

206  ghost_enthalpy = ghost_enthalpy + resid;

237  virtual void solve(Real& ghost_enthalpy,
238  const Real a_interior_temperature,
239  const Real a_ghost_bulk_conc)
240  {
241  CH_TIME("NonlinearBCSolverNewton::solve");
242 
243  // This is somewhat arbitrary
244  Real dH = 0.01;
245 
246  Real res_old = m_max_resid*10, res_new = m_max_resid*10;
247  int num_iter = 0;
248 
249  while (abs(res_old) > m_max_resid && num_iter < m_max_iter)
250  {
251 
252  // Need residual at H and H+dH
253  m_bc->computeResidual(res_old,
254  ghost_enthalpy, // enthalpy in ghost cell
255  a_ghost_bulk_conc, // bulk conc in ghost cell
256  a_interior_temperature // temperature at interior grid cell
257  );
258 
259  m_bc->computeResidual(res_new,
260  ghost_enthalpy+dH, // enthalpy in ghost cell
261  a_ghost_bulk_conc, // bulk conc in ghost cell
262  a_interior_temperature // temperature at interior grid cell
263  );
264 
265  // Near the eutectic, can have the same temperature for different enthalpies and therefore
266  // the same residual. Want to avoid dividing by 0, so just do picard iterations in this case.
267  if (abs(res_new - res_old) < m_max_resid)
268  {
269  ghost_enthalpy = ghost_enthalpy + res_old;
270  // Try increasing dH to stop this happening next time
271  dH = 2*dH;
272  }
273  else
274  {
275  ghost_enthalpy = ghost_enthalpy - res_old * (dH)/(res_new - res_old);
276  }
277 
278  num_iter++;
279  }
280 
281  }
